  pure type(datetime) function ceiling_month(this) result(res)
    class(datetime), intent(in) :: this
    res = datetime(this%year, this%month, 1, 0, 0, 0, calendar=this%calendar)
    call res%add_months(1)
    return
  end function ceiling_month
