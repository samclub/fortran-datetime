  pure integer function days_of_year(year, calendar) result(res)
    integer, intent(in) :: year
    integer, intent(in) :: calendar
    select case (calendar)
    case (gregorian)
      if (is_leap_year(year)) then
        res = 366
      else
        res = 365
      end if
    case (noleap)
      res = 365
    end select
  end function days_of_year
